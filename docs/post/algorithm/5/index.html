【算法系列】用一个栈实现另一个栈的排序

2022年8月1日
| algorithm

题目要求

一个栈中元素的类型为整型，现在想将该栈从顶到底按从小到大的顺序排序，只允许申请一个栈。除此之外，可以申请新的变量，但不能申请额外的数据结构。如何完成排序？

解题思路

  1. 申请一个新的help栈，不断从原有栈中获取数据, 去跟新的help栈的栈顶数据相比; 如果符合排序要求，就push到help栈中
  2. 如果不符合排序要求，就从help栈中pop出栈顶数据，push到原栈中，直至符合排序要求
  3. 最终原栈被清空，help栈是全部排好序的栈，重新一股脑写回来即完成栈排序

golang实现

package main

// SortStark stark top --> stark bottom(from small to big)
func SortStark(stk *Stack) {
	helpStack := NewStack()
	// 1. while go through target stack until stack is empty
	for stk.Len() > 0 {
		cur := stk.Pop()
		// 2. if cur < peek, push helpStack.pop() to target stack until cur > peek or helpStack is empty
		for (helpStack.Len() > 0) && (cur.(int) < helpStack.Peek().(int)) {
			stk.Push(helpStack.Pop())
		}
		// 3. if helkStack is not empty, compare cur with helpStack's peek value
		// 4. while cur > peek, push cur to helpStack
		helpStack.Push(cur)
	}
	// 5. push back helpStack to target stack
	for helpStack.Len() > 0 {
		stk.Push(helpStack.Pop())
	}
}

测试用例

package main

import (
	"testing"

	"github.com/stretchr/testify/assert"
)

func TestSortStark(t *testing.T) {
	tests := []struct {
		// 名字
		name string

		// 输入部分
		val *Stack

		// 输出部分
		wantRes []int
	}{
		{
			name: "sortstark",
			val:  NewStack(),
			// transpose once, then return to the original order of last-in first-out
			wantRes: []int{0, 1, 2, 3, 5, 8},
		},
	}
	for _, tt := range tests {
		t.Run(tt.name, func(t *testing.T) {
			// test case
			stk := tt.val
			stk.Push(2)
			stk.Push(8)
			stk.Push(1)
			stk.Push(5)
			stk.Push(0)
			stk.Push(3)
			SortStark(stk)
			res := []int{}
			for stk.length > 0 {
				res = append(res, tt.val.Pop().(int))
			}
			assert.Equal(t, tt.wantRes, res)
		})
	}
}
