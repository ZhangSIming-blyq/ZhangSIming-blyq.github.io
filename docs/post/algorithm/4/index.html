【算法系列】猫狗队列

2022年6月7日
| algorithm

题目要求

狗和猫都实现了Pet接口，可以用getPetType()查看对应的动物类型; NewDog()、NewCat()可以分别新建狗猫对象。

type Pet interface {
	getPetType() string
}

type fatherPet struct {
	Type string
}

func (fp *fatherPet) getPetType() string {
	return fp.Type
}

type Dog struct {
	fatherPet
}

func NewDog() *Dog {
	fmt.Println("new dog")
	return &Dog{fatherPet{Type: "dog"}}
}

func (dog *Dog) getPetType() string {
	return dog.fatherPet.Type
}

type Cat struct {
	fatherPet
}

func NewCat() *Cat {
	fmt.Println("new cat")
	return &Cat{fatherPet{Type: "cat"}}
}

func (cat *Cat) getPetType() string {
	return cat.fatherPet.Type
}

实现一种狗猫队列的结构，要求如下：

  1. 用户可以调用add方法将cat类或dog类的实例放入队列中；
  2. 用户可以调用pollAll方法，将队列中所有的实例按照队列的先后顺序依次弹出；
  3. 用户可以调用pollDog方法，将队列中dog类的实例按照进队列的先后顺序依次弹出；
  4. 用户可以调用pollCat方法，将队列中cat类的实例按照进队列的先后顺序依次弹出；
  5. 用户可以调用isEmpty方法，检查队列中是否还有dog或cat的实例；
  6. 用户可以调用isDogEmpty方法，检查队列中是否有dog类的实例；
  7. 用户可以调用isCatEmpty方法，检查队列中是否有cat类的实例。

解题思路

依照题意，需要确定队列中可能有dog也可能有cat，用两个队列分别保存dog和cat实例；至于如何判断dog队列的元素是更早入队列还是cat队列的元素是更早入队列需要有一个字段用来存储入队列time(全局唯一)，这里可以封装一个struct来保存pet和time；
那么接下来如果想要看dog，就看dogqueue，如果想看cat就看catqueue；如果想看总队列的顺序，取dogqueue和catqueue二者最新元素的time进行比较得出在总队列的顺序。

golang实现

新创建一个准备放入队列的struct，包括Pet对象和Time字段：

type TimePet struct {
	Pet  Pet
	Time int
}

func NewTimePet(pet Pet, time int) *TimePet {
	return &TimePet{
		Pet:  pet,
		Time: time,
	}
}

func (tp *TimePet) getPet() Pet {
	return tp.Pet
}

func (tp *TimePet) getTime() int {
	return tp.Time
}

那么dogcatqueue主体包括逻辑如下：

type DogCatQueue struct {
	DogQueue *customQueue
	CatQueue *customQueue
	Time     int
}

func NewDogCatQueue() *DogCatQueue {
	return &DogCatQueue{
		DogQueue: newCustomQueue(),
		CatQueue: newCustomQueue(),
		// 全局唯一的Time, 用于标识总顺序
		Time:     0,
	}
}

// 添加的时候比较简单，狗进狗队列，猫进猫队列即可
func (dcqueue *DogCatQueue) add(pet Pet) {
	if pet.getPetType() == "dog" {
		dcqueue.Time++
		dcqueue.DogQueue.Enqueue(NewTimePet(pet, dcqueue.Time))
	} else if pet.getPetType() == "cat" {
		dcqueue.Time++
		dcqueue.CatQueue.Enqueue(NewTimePet(pet, dcqueue.Time))
	} else {
		fmt.Errorf("err, not dog or cat")
	}
}

// pollAll的时候需要对比狗队列和猫队列最新元素的Time谁小，谁小表示谁先进的队列
func (dcqueue *DogCatQueue) pollAll() Pet {
	if !dcqueue.isDogQueueEmpty() && !dcqueue.isCatQueueEmpty() {
		if dcqueue.DogQueue.Front().(*TimePet).getTime() < dcqueue.CatQueue.Front().(*TimePet).getTime() {
			return dcqueue.DogQueue.Dequeue().(*TimePet).getPet()
		} else {
			return dcqueue.CatQueue.Dequeue().(*TimePet).getPet()
		}
	} else if !dcqueue.isDogQueueEmpty() {
		return dcqueue.DogQueue.Dequeue().(*TimePet).getPet()
	} else if !dcqueue.isCatQueueEmpty() {
		return dcqueue.CatQueue.Dequeue().(*TimePet).getPet()
	} else {
		fmt.Errorf("err, dogcatqueue is empty")
		return nil
	}
}

// pollDog直接从狗队列删除即可
func (dcqueue *DogCatQueue) pollDog() Pet {
	if !dcqueue.isDogQueueEmpty() {
		fmt.Println("delete latest dog")
		return dcqueue.DogQueue.Dequeue().(*TimePet).getPet()
	} else {
		fmt.Errorf("err, dogcatqueue don't have dog")
		return nil
	}
}

// 同理
func (dcqueue *DogCatQueue) pollCat() Pet {
	if !dcqueue.isCatQueueEmpty() {
		fmt.Println("delete latest cat")
		return dcqueue.CatQueue.Dequeue().(*TimePet).getPet()
	} else {
		fmt.Errorf("err, dogcatqueue don't have cat")
		return nil
	}
}

func (dcqueue *DogCatQueue) isEmpty() bool {
	return dcqueue.DogQueue.Empty() && dcqueue.CatQueue.Empty()
}

func (dcqueue *DogCatQueue) isDogQueueEmpty() bool {
	return dcqueue.DogQueue.Empty()
}

func (dcqueue *DogCatQueue) isCatQueueEmpty() bool {
	return dcqueue.CatQueue.Empty()
}

运行效果

func main() {
	testqueue := NewDogCatQueue()
	testqueue.add(NewDog())
	testqueue.add(NewDog())
	fmt.Println("DogCatQueue has dog? ", !testqueue.isDogQueueEmpty())
	fmt.Println("DogCatQueue has cat? ", !testqueue.isCatQueueEmpty())
	fmt.Println("DogCatQueue has pet? ", !testqueue.isEmpty())
	fmt.Println()
	testqueue.add(NewCat())
	fmt.Println("DogCatQueue has dog? ", !testqueue.isDogQueueEmpty())
	fmt.Println("DogCatQueue has cat? ", !testqueue.isCatQueueEmpty())
	fmt.Println("DogCatQueue has pet? ", !testqueue.isEmpty())
	fmt.Println()
	testqueue.pollDog()
	fmt.Println("DogCatQueue has dog? ", !testqueue.isDogQueueEmpty())
	fmt.Println("DogCatQueue has cat? ", !testqueue.isCatQueueEmpty())
	fmt.Println("DogCatQueue has pet? ", !testqueue.isEmpty())
	fmt.Println()
	testqueue.pollCat()
	fmt.Println("DogCatQueue has dog? ", !testqueue.isDogQueueEmpty())
	fmt.Println("DogCatQueue has cat? ", !testqueue.isCatQueueEmpty())
	fmt.Println("DogCatQueue has pet? ", !testqueue.isEmpty())
	fmt.Println()
	testqueue.pollDog()
	fmt.Println("DogCatQueue has dog? ", !testqueue.isDogQueueEmpty())
	fmt.Println("DogCatQueue has cat? ", !testqueue.isCatQueueEmpty())
	fmt.Println("DogCatQueue has pet? ", !testqueue.isEmpty())
	fmt.Println()
	testqueue.add(NewDog())
	testqueue.add(NewCat())
	testqueue.pollAll()
	fmt.Println("DogCatQueue has dog? ", !testqueue.isDogQueueEmpty())
	fmt.Println("DogCatQueue has cat? ", !testqueue.isCatQueueEmpty())
	fmt.Println("DogCatQueue has pet? ", !testqueue.isEmpty())
	fmt.Println()
	testqueue.pollAll()
	testqueue.add(NewCat())
	testqueue.add(NewDog())
	testqueue.pollAll()
	fmt.Println("DogCatQueue has dog? ", !testqueue.isDogQueueEmpty())
	fmt.Println("DogCatQueue has cat? ", !testqueue.isCatQueueEmpty())
	fmt.Println("DogCatQueue has pet? ", !testqueue.isEmpty())
	fmt.Println()
}

补充: 使用golang链表实现queue

其实使用slice也可以实现，但是如果slice满了以后就会对底层数组进行一次拷贝, 使用链表则没有这个问题, 实现如下(接受interface{}作为队列元素)：

package main

import (
	"container/list"
)

type customQueue struct {
	queue *list.List
}

func newCustomQueue() *customQueue {
	return &customQueue{queue: list.New()}
}

func (c *customQueue) Enqueue(value interface{}) {
	c.queue.PushBack(value)
}

func (c *customQueue) Dequeue() interface{} {
	if c.queue.Len() > 0 {
		ele := c.queue.Front()
		c.queue.Remove(ele)
		return ele.Value
	}
	return nil
}

func (c *customQueue) Front() interface{} {
	if c.queue.Len() > 0 {
		return c.queue.Front().Value
	}
	return nil
}

func (c *customQueue) Size() int {
	return c.queue.Len()
}

func (c *customQueue) Empty() bool {
	return c.queue.Len() == 0
}
