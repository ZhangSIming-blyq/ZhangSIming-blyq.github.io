【算法系列】如何仅用递归函数和栈操作逆序一个栈

2022年6月5日
| algorithm

题目要求

一个栈依次压入1、2、3、4、5，那么从栈顶到栈底分别为5、4、3、2、1。将这个栈转置后，从栈顶到栈底为1、2、3、4、5，也就是实现栈元素的逆序，但是只能用递归函数来实现，不能用其他数据结构。

解题思路

  1. 一个函数实现返回栈内数据，用于返回栈底元素，当栈空的时候停止
  2. 另一个函数接受栈底元素并将每一个的数据重新压入栈中，即可实现逆序

golang实现

package main

import "fmt"

func getAndRemoveLastElement(stack *Stack) int {
	result := stack.Pop()
	if stack.Len() == 0 {
		return result.(int)
	} else {
		last := getAndRemoveLastElement(stack)
		stack.Push(result)
		return last
	}
}

func reverse(stack *Stack) {
	if stack.Len() == 0 {
		return
	} else {
		garFuncReturn := getAndRemoveLastElement(stack)
		reverse(stack)
		stack.Push(garFuncReturn)
	}
}

测试用例

package main

import (
	"testing"

	"github.com/stretchr/testify/assert"
)

func TestReverseStack(t *testing.T) {
	// define a getmin stack
	tests := []struct {
		// 名字
		name string

		// 输入部分
		val func(stk *Stack)

		// 输出部分
		wantRes []any
	}{
		{
			name: "push",
			val:  push,
			// transpose once, then return to the original order of last-in first-out
			wantRes: []any{1, 2, 3, 4, 5},
		},
	}
	for _, tt := range tests {
		// new TwoStackQueue
		stk := NewStack()
		t.Run(tt.name, func(t *testing.T) {
			tt.val(stk)
			reverse(stk)
			res := popall(stk)
			assert.Equal(t, tt.wantRes, res)
		})
	}
}

func push(stk *Stack) {
	for i := 1; i <= 5; i++ {
		stk.Push(i)
	}
}

func popall(stk *Stack) (resultList []any) {
	for {
		if stk.length == 0 {
			break
		}
		resultList = append(resultList, stk.Pop())
	}
	return resultList
}
